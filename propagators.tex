\begin{frame}[fragile]
\centering \huge
Bounded join semilattice
\nl
\large

Identity: \\
$x \vee bottom = bottom = bottom \vee x$
\nl

Associative: \\
$x \vee (y \vee z) = (x \vee y) \vee z$
\nl

Commutative: \\
$x \vee y = y \vee x$
\nl

Idempotent: \\
$x \vee x = x$

\end{frame}

class SemiLattice a where

\begin{frame}[fragile]
\begin{haskellcode}
-- types
data Cell a

data Par a
instance Monad Par
\end{haskellcode}
\pnl
\begin{haskellcode}
-- Creating a cell
cell    :: Par (Cell a)
\end{haskellcode}
\pnl
\begin{haskellcode}
-- Working with Cells
content :: Cell a -> Par (Maybe a)
write   :: Cell a -> a -> Par ()
\end{haskellcode}
\pnl
\begin{haskellcode}
-- Creating a propagator
watch   :: Cell a -> (a -> Par ()) -> Par ()
\end{haskellcode}
\end{frame}
